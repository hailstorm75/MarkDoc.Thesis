\chapter*{Glossary}

\begin{multicols}{2}
    \raggedright
    \begin{description}
        \item [\namedlabel{gloss:dotnetlabel}{.NET}] A free, cross-platform, open source developer platform for building many different types of applications
        \item [\namedlabel{gloss:aspnetcore}{ASP.NET Core}] A cross-platform, high-performance, open-source framework for building modern, cloud-enabled, Internet-connected apps
        \item [\namedlabel{gloss:nuget}{NuGet}] Package manager for .NET projects
    \end{description}
